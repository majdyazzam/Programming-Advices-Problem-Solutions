int GetRandomNumber(int first,int last) {

	int result;
	return result = rand() % (last - first + 1) + first;
}

void fillArrayRandomNumber(int arr1[3][3],int row,int col) {
	for (int i = 0; i < row;i++) {
		for (int j = 0; j < col;j++) {
			arr1[i][j] = GetRandomNumber(0, 100);
		}
	}
}

void Print2DArray(int arr1[3][3], int row, int col) {

	for (int i = 0; i < row; i++) {
		for (int j = 0; j < col; j++) {
			//cout  << arr1[i][j] << setw(5);
			printf("%0*d   ",2,arr1[i][j]);
		}
		cout << "\n";
	}
}

		//cout << arr1[i][j] << "  ";

}
