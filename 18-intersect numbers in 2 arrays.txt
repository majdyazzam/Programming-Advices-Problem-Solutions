int GenerateRandNumber(int First,int Last) {
	int number;
	number = rand() % (Last - First + 1) + First;
		return number;
}

void FillRandNumberInArray(int arr[3][3],int row,int col) {

	for (int i = 0; i < row;i++) {
		for (int j = 0; j < col;j++) {
			arr[i][j] = GenerateRandNumber(1,100);
		}
	}
}

void Print2DArray(int arr1[3][3], int row, int col) {

	for (int i = 0; i < row; i++) {
		for (int j = 0; j < col; j++) {
			//cout  << arr1[i][j] << setw(5);
			printf("%0*d   ",2,arr1[i][j]);
		}
		cout << "\n";
	}
}

bool FindNumInArray(int arr1[3][3],int row,int col,int num) {

	for (int i = 0; i < row;i++) {
		for (int j = 0; j < col;j++) {
			if (arr1[i][j] == num)
				return true;
		}
	}
	return false;
}

void printNumberInArray(int arr1[3][3],int arr2[3][3], int row, int col) {
	int num;
	for (int i = 0; i < row; i++) {
		for (int j = 0; j < col; j++) {
			num = arr1[i][j];
			if (FindNumInArray(arr2, row, col, num))
				cout << num << "    ";
		}
	}
	cout << "\n";
}
