int	 ReadPositiveNumber() {
	int num;
	do {
		
		cin >> num;
	} while (num < 0);

	return num;
}


void Print2DArray(int arr1[3][3], int row, int col) {

	for (int i = 0; i < row; i++) {
		for (int j = 0; j < col; j++) {
			//cout  << arr1[i][j] << setw(5);
			printf("%0*d   ",2,arr1[i][j]);
		}
		cout << "\n";
	}
}


int CountNumber(int arr[3][3],int row,int col,int num) {

	int count = 0;
	for (int i = 0; i < row; i++) {
		for (int j = 0; j < col; j++) {
			if (arr[i][j]==num) {
				count++;
			}
		}
	}
	return count;
}
int main()
{

	
	int arr1[3][3] = { {7,0,0},{0,7,0},{0,0,7} };

	Print2DArray(arr1, 3, 3);

	cout << "Enter number you want to count : ";
	int num = ReadPositiveNumber();

	cout<< "the number of "<<num <<" apears " << CountNumber(arr1, 3, 3, num)<<" Times\n";
	
	return 0;
}
