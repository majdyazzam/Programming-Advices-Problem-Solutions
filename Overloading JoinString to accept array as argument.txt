string JoinString(vector <string> VStr,string separator) {
	string JoinStr;
	for (short i = 0; i < VStr.size();i++) {
		if (i != VStr.size()-1)
			JoinStr += VStr.at(i) + separator;
		else
			JoinStr += VStr.at(i);
	}
	return JoinStr;
