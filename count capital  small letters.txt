enum enWhatToCount { smallLetter = 0, CapitalLetter = 1,All=2 };

short CountLetter(string txt, enWhatToCount whatCount) {
	short counter = 0;

	for (short i = 0; i < txt.length(); i++)
	{
		if (whatCount == enWhatToCount::All) {
			return txt.length();
		}

		if (whatCount== enWhatToCount::CapitalLetter && isupper(txt[i]))
			counter++;
		if (whatCount == enWhatToCount::smallLetter && islower(txt[i]))
			counter++;
	}
	return counter;
}

short CountCapitalLetters(string txt) {
	short UpperCount = 0;

	for (short i = 0; i < txt.length(); i++)
	{
		if (isupper(txt[i]))
			UpperCount++;
		
	}
	return UpperCount;
}

short CountSmallLetters(string txt) {
	short  LowerCount = 0;

	for (short i = 0; i < txt.length(); i++)
	{
		if (islower(txt[i]))
			LowerCount++;

	}
	return LowerCount;
}
