char ReadChar() {
	char txt;
	cout << "\nPlease Enter Your Character : \n";
	cin >> txt;
	return txt;
}


short CountCharInLetters(string txt,char ch) {
	short  Count = 0;

	for (short i = 0; i < txt.length(); i++)
	{
		if ((txt[i])==ch)
			Count++;

	}
	return Count;
}

{
	string s1 = ReadString();
	char ch = ReadChar();
