

#include <iostream>
#include<iomanip>
#include<string>
using namespace std;
int GetRandomNumber(int first,int last) {
	return rand() % (last - first + 1) + first;
}

void FillRandomNumberInTOArray(int arr[3][3],short rows,short col) {
	

	for (int i = 0; i < rows; i++) {
			for (int j = 0; j <= col;j++) {

				arr[i][j] = GetRandomNumber(1, 100);
			}
		}
}

void printArray(int arr[3][3],int rows,int col) {
	
	for (int i = 0; i < rows;i++) {
		for (int j = 0; j <= col;j++) {
			
			cout  << setw(5) << arr[i][j]<<"   ";
			
		}
		cout << endl;
	}
}
int main()
{
	srand((unsigned) time(NULL));
	int arr[3][3];
	FillRandomNumberInTOArray(arr, 3, 3);
	cout << "\nThe Follwing is a 3  X 3 random matrix :\n";
	printArray(arr,3,3);

	system("pause>0");
}
