

#include <iostream>

using namespace std;



void Print2DArray(int arr1[3][3], int row, int col) {

	for (int i = 0; i < row; i++) {
		for (int j = 0; j < col; j++) {
			//cout  << arr1[i][j] << setw(5);
			printf("%0*d   ",2,arr1[i][j]);
		}
		cout << "\n";
	}
}


bool CheckScalarity(int arr1[3][3],int row, int col) {
	int temp = arr1[0][0];
	for (int i = 0; i < row;i++) {
		for (int j = 0; j < col;j++) {

			if (i == j && arr1[i][j] != temp || temp ==1) {
				return false;
			}

			
			else if (i != j && arr1[i][j] != 0) {
				return false;
			}
				
		}
		
	}
	return  true;
}

int main()
{

	
	int arr1[3][3] = { {7,0,0},{0,7,0},{0,0,7} };

	Print2DArray(arr1, 3, 3);

	if (CheckScalarity(arr1, 3, 3))
		cout << "\nYes : both matrix is Scalar \n";
	else
		cout << "\nNo : both matrix is NOT Scalar \n";
	
	return 0;
}
