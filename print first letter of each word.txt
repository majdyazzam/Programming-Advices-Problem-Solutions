void FirstLetterOfEachWord(string txt) {

	string tempTxt = "";
	for (int i = 0; i < txt.length(); i++) {
		if (txt[i]==' ')
		{
			cout << tempTxt[0] << endl;
			tempTxt = "";
			i++;
		}
		
			tempTxt += txt[i];
	
	}
	
	cout << tempTxt[0] << endl;
}

void PrintFirstLetterOfEachWord(string txt) {
	bool isFirstLetter = true;

	for (short i = 0; i < txt.length(); i++)
	{
		if (txt[i] != ' ' && isFirstLetter) {
			cout << txt[i] << endl;
		}
		isFirstLetter = (txt[i] == ' ' ? true : false);
	}

}
