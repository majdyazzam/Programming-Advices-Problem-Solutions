int ReadPositiveNumber() {
	int num;
	do {
		cout << "Enter Positive Number : ";
		cin >> num;
	} while (num < 0);
	return num;
}

void Print2DArray(int arr1[3][3], int row, int col) {

	for (int i = 0; i < row; i++) {
		for (int j = 0; j < col; j++) {
			//cout  << arr1[i][j] << setw(5);
			printf("%0*d   ",2,arr1[i][j]);
		}
		cout << "\n";
	}
}



bool CheckNumberInMatrix(int arr[3][3], int row, int col,int num) {
	for (int i = 0; i < row;i++) {
		for (int j = 0; j < col;j++) {
			if (arr[i][j] == num)
				return true;
		}
	}
	return false;
}
int main()
{

	
	int arr1[3][3] = { {7,0,5},{4,7,0},{1,0,7} };

	Print2DArray(arr1, 3, 3);

	int num = ReadPositiveNumber();
	if (CheckNumberInMatrix(arr1, 3, 3,num))

		cout << "Yes : It is there .\n";
	else
		cout<< "No : It is Not there .\n";
	return 0;
}
