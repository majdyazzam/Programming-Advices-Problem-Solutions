

#include <iostream>
#include<iomanip>
#include<string>
using namespace std;

int GetRandomNumber(int first,int last) {

	int result;
	return result = rand() % (last - first + 1) + first;
}

void fillArrayRandomNumber(int arr1[3][3],int row,int col) {
	for (int i = 0; i < row;i++) {
		for (int j = 0; j < col;j++) {
			arr1[i][j] = GetRandomNumber(0, 100);
		}
	}
}

void Print2DArray(int arr1[3][3], int row, int col) {

	for (int i = 0; i < row; i++) {
		for (int j = 0; j < col; j++) {
			//cout  << arr1[i][j] << setw(5);
			printf("%0*d   ",2,arr1[i][j]);
		}
		cout << "\n";
	}
}



bool CheckTypical(int arr1[3][3],int arr2[3][3],int row, int col) {

	for (int i = 0; i < row;i++) {
		for (int j = 0; j < col;j++) {
			if (arr1[i][j] != arr2[i][j])
				return 0;
		}
	}
	return 1;
}

int main()
{
	srand((unsigned)time(NULL));
	
	int arr1[3][3];
	int arr2[3][3];

	fillArrayRandomNumber(arr1,3,3);
	fillArrayRandomNumber(arr2, 3, 3);

	cout << "Matrix 1 : \n";
	Print2DArray(arr1,3,3);

	cout << "Matrix 2 : \n";
	Print2DArray(arr2, 3, 3);

	if (CheckTypical(arr1, arr2, 3, 3))
		cout << "\nNo : both matricies are NOT Typical \n";
	else
		cout << "\nYes : both matricies are  Typical \n";
	
	system("pause>0");
	return 0;
}
