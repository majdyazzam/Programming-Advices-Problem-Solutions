string ConvertFirstLetterToUpper(string txt) {

	bool isFirstLetter = true;
	
	for (short i = 0; i < txt.length(); i++)
	{
		if (txt[i] != ' '&& isFirstLetter) {
			txt[i]=toupper(txt[i]);
		}
		
		isFirstLetter = (txt[i] == ' ' ? true : false);
		
	}
	return txt;
}

int main()
{
	string s1 = ReadString();

	s1=	ConvertFirstLetterToUpper(s1);
