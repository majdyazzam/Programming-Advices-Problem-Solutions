

#include <iostream>
#include<iomanip>
#include<string>
using namespace std;

int GetRandomNumber(int first,int last) {

	int result;
	return result = rand() % (last - first + 1) + first;
}

void fillArrayRandomNumber(int arr1[3][3],int row,int col) {
	for (int i = 0; i < row;i++) {
		for (int j = 0; j < col;j++) {
			arr1[i][j] = GetRandomNumber(0, 100);
		}
	}
}

void MultiplyArr1WithArr2(int arr1[3][3], int arr2[3][3],
	int arrResult[3][3], int row, int col) {

	for (int i = 0; i < row; i++) {
		for (int j = 0; j < col; j++) {
			arrResult[i][j] = arr1[i][j] * arr2[i][j];
		}
	}
}

void Print2DArray(int arr1[3][3], int row, int col) {

	for (int i = 0; i < row; i++) {
		for (int j = 0; j < col; j++) {
			//cout  << arr1[i][j] << setw(5);
			printf("%0*d   ",2,arr1[i][j]);
		}
		cout << "\n";
	}
}

int main()
{
	srand((unsigned)time(NULL));
	
	int arr1[3][3];
	int arr2[3][3];
	int arr3[3][3];

	fillArrayRandomNumber(arr1,3,3);
	fillArrayRandomNumber(arr2,3,3);

	MultiplyArr1WithArr2(arr1, arr2, arr3, 3, 3);

	cout << "Matrix 1 : \n";
	Print2DArray(arr1,3,3);

	cout << "\nMatrix 2 : \n";
	Print2DArray(arr2, 3, 3);

	cout << "\nResults : \n";
	Print2DArray(arr3, 3, 3);

	system("pause>0");
	return 0;
}
